解构赋值 ​

1、数组解构

js
// 具有Iterator接口的数据结构
const [a, [[b],c]] = [1, [[2], 3]];
console.log(a, b, c) // 1, 2, 3

// 不完全解构

js
const [,a,b] = [1,2,3];
console.log(a, b) // 2, 3

// 剩余运算符

js
const [a, ...b] = [1,2,3]
console.log(a, b) // 1, [2, 3]

// 解构赋值的默认值

js
const [a,b,c=3] = [1,2]
console.log(a, b, c) // 1, 2, 3

2、对象解构

js
const {a, b, c} = {a: 1, b: 2, c: 3}
console.log(a, b, c) // 1, 2, 3

// 剩余运算符

js
const {a,...b} = {a: 1, b: 2, c: 3}
console.log(a, b) // 1, {b: 2, c: 3}

3、其他赋值解构

  • 字符串解构
js
const [a, b, c, d, e] = 'hello'
console.log(a, b, c, d, e) // h, e, l, l, o

const {length: len} = 'hello'
console.log(len) // 5
